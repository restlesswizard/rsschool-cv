Example Three

Task RGB to HEX (codewars.com): write a function, that will get RGB decimal values and will return its hexadecimal representation.

Condition: RGB decimal values, passing into function, may be out of range 0 to 255. 

function rgb(r, g, b) {
	let decimals = Object.values(arguments);

	let answer = ''

	decimals.forEach(function(elem, index) {
		if (elem < 0) {
			this[index] = 0;
		} else if (elem > 255) {
			this[index] = 255;
		}
	}, decimals)

	decimals.map((digit) => {
		if (digit < 16) {
			answer += '0' + digit.toString(16).toUpperCase();
		} else {
			answer += digit.toString(16).toUpperCase();
		}
	})

	return answer
}
